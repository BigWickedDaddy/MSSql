CREATE TRIGGER ID_DP ON Directory_of_parts

AFTER UPDATE AS

BEGIN

UPDATE PARTS_PROFESSION

SET TARIF_COD= (SELECT TARIF_COD FROM INSERTED) WHERE PARTS_PROFESSION.KOD_PARTS=(SELECT KOD_PARTS FROM INSERTED)

END

--------------

CREATE TRIGGER ID_Directory_of_parts ON Directory_of_parts

AFTER INSERT AS

BEGIN

UPDATE PARTS_PROFESSION

SET TARIF_COD =TARIF_COD + (select SUM(PLAN_PRICE) from inserted
			WHERE inserted.KOD_PARTS = PARTS_PROFESSION.KOD_PARTS)
WHERE KOD_PARTS in (select KOD_PARTS from inserted) 

END
